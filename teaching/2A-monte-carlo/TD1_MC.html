import numpy as np

def randu(x0, n):
    """
    input :
      x0 une valeur deterministe a fixer, 
      n la taille de la liste que l on souhaite avoir
    output :
      u = (u_1, ..., u_n)
    """
    U = []
    x = x0
    for i in range(n):
        a = (2 ** 16 + 3) * x
        x = a % (2 ** 31)
        U.append(x / (2 ** 31) )
    U = np.array(U)
    return U
